pipeline {
    agent any

    stages {
        stage('Checkout ') {
            steps {
                checkout([$class: 'GitSCM',
                branches: [[name: '*/master']], 
                doGenerateSubmoduleConfigurations: false,
                extensions: [[$class: 'RelativeTargetDirectory', 
                relativeTargetDir: 'akash_scripts']], 
                submoduleCfg: [], 
                userRemoteConfigs: [[url: 'https://github.com/Akashbhaskaran/scripts']]])

            }
            
        }
        stage('Executing') {
         steps {   
             
            sh "chmod  +x akash_scripts/task1.sh"
            sh "chmod  +x akash_scripts/task2.sh"
          
          sh('akash_scripts/task1.sh')   
          sh('akash_scripts/task2.sh')   
         }
        }
        stage('Archiving'){
         
         steps {
          archiveArtifacts "akash_scripts/*.sh"   
         }
         }
         stage('Final stage'){
            steps {
                echo "RESULT: ${currentBuild.currentResult}"
            } 
         } 
         
            
        }
		post {
        always {
       echo "RESULT: ${currentBuild.currentResult}"
    }
    }
        }
